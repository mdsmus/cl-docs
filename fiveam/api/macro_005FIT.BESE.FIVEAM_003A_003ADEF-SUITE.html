(defmacro def-suite (name &key description in)
  "Define a new test-suite named NAME.

IN (a symbol), if provided, causes this suite te be nested in the
suite named by IN."
  `(progn
     (make-suite ',name
		 ,@(when description `(:description ,description))
		 ,@(when in `(:in ',in)))
     ',name))
